type Zipper&lt;’a&gt; = Zipper of (‘a list * ‘a * ‘a list) with
    static member create l =
        match l with
        | [] -&gt; failwith “oops”
        | h :: t -&gt; Zipper ([], h, t)</p>

<pre><code>member z.right () =
    match z with
    | Zipper(l, z, []) -&gt; Zipper(l, z, [])
    | Zipper(l, z, h::rt) -&gt; Zipper(z::l, h, rt)

member z.left () =
    match z with
    | Zipper([], z, r) -&gt; Zipper([], z, r)
    | Zipper(h::lt, z, r) -&gt; Zipper(lt, h, z::r)

member z.update x =
    let (Zipper(l, _, r)) = z in Zipper(l, x, r)
