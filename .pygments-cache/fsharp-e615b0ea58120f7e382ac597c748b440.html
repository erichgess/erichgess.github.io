let rec forall f m = 
    match m with 
    | SetNode(k2,l,r,_) -> f k2 && forall f l && forall f r
    | SetOne(k2) -> f k2
    | SetEmpty -> true          

let rec exists f m = 
    match m with 
    | SetNode(k2,l,r,_) -> f k2 || exists f l || exists f r
    | SetOne(k2) -> f k2
    | SetEmpty -> false         

let subset comparer a b  = forall (fun x -> mem comparer x b) a
