let rec insert3 (x, prev: 'a option, t) =
  match t with
  | Empty -> if prev.IsSome && x = prev.Value 
             then failwith "Element already exists" 
             else Tree (Empty, x, Empty)
  | Tree(l, n, r) ->
    if x < n then Tree(insert3 (x, prev, l), n, r)
    else Tree(l, n, insert3 (x, Some n, r)
