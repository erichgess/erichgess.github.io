let mk l k r = 
    match l,r with 
    | SetEmpty,SetEmpty -&gt; SetOne (k)
    | _ -&gt; 
      let hl = height l 
      let hr = height r 
      let m = if hl &lt; hr then hr else hl 
      SetNode(k,l,r,m+1)</p>

<p>let rebalance t1 k t2 =
    let t1h = height t1 
    let t2h = height t2 
    if  t2h &gt; t1h + tolerance then // right is heavier than left 
        match t2 with 
        | SetNode(t2k,t2l,t2r,<em>) -&gt; 
            // one of the nodes must have height &gt; height t1 + 1 
            if height t2l &gt; t1h + 1 then  // balance left: combination 
                match t2l with 
                | SetNode(t2lk,t2ll,t2lr,</em>) -&gt;
                    mk (mk t1 k t2ll) t2lk (mk t2lr t2k t2r) 
                | _ -&gt; failwith “rebalance”
            else // rotate left 
                mk (mk t1 k t2l) t2k t2r
        | _ -&gt; failwith “rebalance”
    else
        if  t1h &gt; t2h + tolerance then // left is heavier than right 
            match t1 with 
            | SetNode(t1k,t1l,t1r,<em>) -&gt; 
                // one of the nodes must have height &gt; height t2 + 1 
                if height t1r &gt; t2h + 1 then 
                    // balance right: combination 
                    match t1r with 
                    | SetNode(t1rk,t1rl,t1rr,</em>) -&gt;
                        mk (mk t1l t1k t1rl) t1rk (mk t1rr k t2)
                    | _ -&gt; failwith “rebalance”
                else
                    mk t1l t1k (mk t1r k t2)
            | _ -&gt; failwith “rebalance”
        else mk t1 k t2
