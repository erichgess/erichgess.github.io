let rec remove (comparer: IComparer&lt;’T&gt;) k t = 
    match t with 
    | SetEmpty -&gt; t
    | SetOne (k2) -&gt; 
        let c = comparer.Compare(k,k2) 
        if   c = 0 then SetEmpty
        else t
    | SetNode (k2,l,r,<em>) -&gt; 
        let c = comparer.Compare(k,k2) 
        if   c &lt; 0 then rebalance (remove comparer k l) k2 r
        elif c = 0 then 
          match l,r with 
          | SetEmpty,</em> -&gt; r
          | _,SetEmpty -&gt; l
          | _ -&gt; 
              let sk,r’ = spliceOutSuccessor r 
              mk l sk r’
        else rebalance l k2 (remove comparer k r)
