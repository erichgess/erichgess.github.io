let rec mem (comparer: IComparer<'T>) k t = 
    match t with 
    | SetNode(k2,l,r,_) -> 
        let c = comparer.Compare(k,k2) 
        if   c < 0 then mem comparer k l
        elif c = 0 then true
        else mem comparer k r
    | SetOne(k2) -> (comparer.Compare(k,k2) = 0)
    | SetEmpty -> false
